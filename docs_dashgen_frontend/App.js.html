import React, { useState } from 'react';
import {
  BrowserRouter as Router,
  Switch,
  Route,
} from 'react-router-dom';
import Home from './pages/Home/Home';
import Panel from './pages/Panel/Panel';
import ExportData from './pages/ExportData/ExportData';
import Variables from './pages/Variables/Variables';
import Summary from './pages/Summary/Summary';

/**
 * Componente principal para gerenciar rotas
 *
 * @component
 */

function App() {
  const [selectedPanel, setSelectedPanel] = useState([]);
  const [selectedVariable, setSelectedVariable] = useState({});
  return (
    <Router>
      <Switch>
        <Route path="/selecao-paineis">
          <Panel selectedPanel={selectedPanel} setSelectedPanel={setSelectedPanel} />
        </Route>
        <Route path="/selecao-variaveis">
          <Variables
            selectedPanel={selectedPanel}
            selectedVariable={selectedVariable}
            setSelectedVariable={setSelectedVariable}
          />
        </Route>
        <Route path="/resumo">
          <Summary selectedVariable={selectedVariable} />
        </Route>
        <Route path="/exportar-dados">
          <ExportData />
        </Route>
        <Route path="/">
          <Home />
        </Route>
      </Switch>
    </Router>
  );
}

export default App;
