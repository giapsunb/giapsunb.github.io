import React from 'react';
import { Link } from 'react-router-dom';
import Grid from '@material-ui/core/Grid';
import Button from '../../components/Button/Button';
import Navbar from '../../components/Navbar/Navbar';
import './Summary.css';

function sortText(a, b) {
  return a.localeCompare(b);
}

/**
 * Na Página de resumo, o usuário tem um resumo dos painéis
 * e das variáveis selecionadas.
 *
 * @interface
 *
 */

function Summary({ selectedVariable }) {
  const panel = Object.keys(selectedVariable).sort(sortText);
  return (
    <>
      {/* Navbar div */}
      <Grid item xs={12} sm={12} md={12} lg={12}>
        <Navbar />
      </Grid>

      <Grid container className="contentBox">
        <Grid item xs={12} sm={12} md={12} lg={12}>
          <h2 className="page-title">Resumo da exportação</h2>
        </Grid>

        <Grid container className="summary-text-box">
          <Grid item className="summary-list-text-box" xs={12} sm={12} md={6} lg={6}>
            <h3>Painéis Selecionados:</h3>
            {
              panel.map((item, index) => (
                <p key={index}>{item}</p>
              ))
            }
          </Grid>
          <Grid item className="summary-list-text-box" xs={12} sm={12} md={6} lg={6}>
            <h3>Variáveis selecionadas:</h3>
            {
              panel.map((item, index) => (
                <p key={index}>
                  {selectedVariable[item].sort(sortText).join(' ')}
                </p>
              ))
            }
          </Grid>
        </Grid>

        <Grid className="page-footer" item xs={12} sm={12} md={12} lg={12}>
          <Link to="selecao-variaveis"><Button>ETAPA ANTERIOR</Button></Link>
          <Link to="exportar-dados"><Button>PRÓXIMA ETAPA</Button></Link>
        </Grid>
      </Grid>
    </>
  );
}

export default Summary;
