import React, { useState } from 'react';
import { Link } from 'react-router-dom';
import Grid from '@material-ui/core/Grid';
import Navbar from '../../components/Navbar/Navbar';
import Button from '../../components/Button/Button';
import IconSelectSend from '../../assets/iconSelectSend.svg';
import './Panel.css';

function sortText(a, b) {
  return a.localeCompare(b);
}

/**
 * Na Página de escolha de painéis, o usuário deve escolher
 * os indices fonte que ele deseja obter os dados
 *
 * @interface
 *
 */

function Panel({ selectedPanel, setSelectedPanel }) {
  const [availablePanel, setAvailablePanel] = useState([
    'Saúde da Família',
    'Saúde Bucal',
    'ACS',
    'Situação do prontuário eletrônico',
    'Evolução do prontuário eletrônico',
    'Pessoas Cadastradas',
    'Situação de cadastro',
    'LAI - Lei de Acesso a Informação',
    'Desempenho Geral',
  ].filter((i) => selectedPanel.indexOf(i) === -1).sort(sortText));

  const onClickLeftHandler = (titleItem) => {
    const elementList = [...availablePanel];
    const indexElement = elementList.indexOf(titleItem);
    elementList.splice(indexElement, 1);
    setAvailablePanel(elementList);
    setSelectedPanel([...selectedPanel, titleItem].sort(sortText));
  };

  const onClickRightHandler = (titleItem) => {
    const elementList = [...selectedPanel];
    const indexElement = elementList.indexOf(titleItem);
    elementList.splice(indexElement, 1);
    setSelectedPanel(elementList);
    setAvailablePanel([...availablePanel, titleItem].sort(sortText));
  };

  return (
    <>
      {/* Navbar div */}
      <Grid item xs={12} sm={12} md={12} lg={12}>
        <Navbar />
      </Grid>

      <Grid container className="contentBox">
        <Grid item xs={12} sm={12} md={12} lg={12}>
          <h2 className="page-title">Escolha os painéis</h2>
        </Grid>

        <Grid item className="panel-dualbox-left" xs={12} sm={12} md={5} lg={5}>
          <h3 className="panel-subtitle">Painéis disponíveis</h3>
          <div className="panel-dualbox-content panel-selection-box">
            {
              availablePanel.map((item, index) => <div onClick={() => onClickLeftHandler(item)} key={index} className="panel-dualbox-item"><h4>{item}</h4></div>)
            }
          </div>
        </Grid>
        <Grid item className="panel-dualbox-center" xs={12} sm={12} md={2} lg={2}>
          <img className="panel-icon-select-send" src={IconSelectSend} alt="Enviar para" />
        </Grid>

        <Grid item className="panel-dualbox-right" xs={12} sm={12} md={5} lg={5}>
          <h3 className="panel-subtitle">Painéis selecionados</h3>
          <div className="panel-dualbox-content panel-selected-box">
            {
              selectedPanel.map((item, index) => <div onClick={() => onClickRightHandler(item)} key={index} className="panel-dualbox-item"><h4>{item}</h4></div>)
            }
          </div>
        </Grid>
        <Grid className="page-footer" item xs={12} sm={12} md={12} lg={12}>
          <Link to="/"><Button>ETAPA ANTERIOR</Button></Link>
          <Link to="selecao-variaveis"><Button disabled={selectedPanel.length === 0}>PRÓXIMA ETAPA</Button></Link>
        </Grid>
      </Grid>
    </>
  );
}

export default Panel;
