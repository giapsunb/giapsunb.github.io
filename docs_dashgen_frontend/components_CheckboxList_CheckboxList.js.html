import React from 'react';
import './CheckboxList.css';

/**
 * Retorna uma caixa de seleção
 *
 * @component
 * @example
 * const children = 'ACS'
 * return (
 *   <CheckboxList title="ACS" />
 * )
 */

function CheckboxList({
  titlePanel, variablesList, selectedVariable, setSelectedVariable,
}) {
  const checkedHandle = (event) => {
    event.persist();
    const { checked, value } = event.target;
    let newState;
    if (checked) {
      newState = {
        ...selectedVariable,
        [titlePanel]: selectedVariable[titlePanel]
          ? [...selectedVariable[titlePanel], value] : [value],
      };
    } else {
      const elementList = selectedVariable[titlePanel];
      const indexElement = elementList.indexOf(value);
      elementList.splice(indexElement, 1);
      newState = {
        ...selectedVariable,
        [titlePanel]: elementList,
      };
      if (newState[titlePanel].length === 0) { delete newState[titlePanel]; }
    }
    setSelectedVariable({ ...newState });
  };

  return (
    <>
      <div className="checkbox-list-content">
        <div className="checkbox-list-header">
          <h3 className="checkbox-list-title">{titlePanel}</h3>
        </div>
        <div className="checkbox-list-items">
          {
            variablesList.map((item, index) => (
              <label key={index}>
                <div className="checkbox-list-item">
                  <input className="checkbox-list-input" type="checkbox" checked={selectedVariable[titlePanel] ? selectedVariable[titlePanel].indexOf(item) !== -1 : false} value={item} onChange={checkedHandle} />
                  <h4>{item}</h4>
                </div>
              </label>
            ))
          }
        </div>
      </div>
    </>
  );
}

export default CheckboxList;
