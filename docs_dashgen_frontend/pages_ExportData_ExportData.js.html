import React, { useState } from 'react';
import { Link } from 'react-router-dom';
import Grid from '@material-ui/core/Grid';
import Button from '../../components/Button/Button';
import Navbar from '../../components/Navbar/Navbar';
import './ExportData.css';

/**
 * Na Página de exportação de dados ocorre final do processo. Aqui o usuário
 * deve escolher o formato de exportação e, logo após a escolha, o download será
 * iniciado.
 *
 * @interface
 *
 */

function ExportData() {
  const [selectedFormat, setSelectedFormat] = useState('csv');
  return (
    <>
      <Grid container className="mainContainer-export">
        {/* Navbar div */}
        <Grid item xs={12} sm={12} md={12} lg={12}>
          <Navbar />
        </Grid>

        <Grid container className="contentBox">
          <Grid item m={12} md={12} lg={12}>
            <h2 className="page-title">Escolha o formato de Saída</h2>
          </Grid>

          <Grid container className="export-content">
            <Grid item className="export-item-box" xs={false} sm={false} md={3} lg={2} />
            <Grid item className="export-item-box export-item-box-first" xs={12} sm={12} md={3} lg={4}>
              <div
                className={`export-item ${selectedFormat === 'csv' ? 'export-format-selected' : ''}`}
                onClick={() => { setSelectedFormat('csv'); }}
              >
                CSV

              </div>
            </Grid>
            <Grid item className="export-item-box" xs={12} sm={12} md={3} lg={4}>
              <div
                className={`export-item ${selectedFormat === 'json' ? 'export-format-selected' : ''}`}
                onClick={() => { setSelectedFormat('json'); }}
              >
                JSON

              </div>
            </Grid>
            <Grid item className="export-item-box" xs={false} sm={false} md={3} lg={2} />
          </Grid>

          <Grid className="page-footer" item xs={12} sm={12} md={12} lg={12}>
            <Link to="resumo"><Button>ETAPA ANTERIOR</Button></Link>
            <Link to="exportar-dados"><Button>EXPORTAR</Button></Link>
          </Grid>
        </Grid>
      </Grid>
    </>
  );
}

export default ExportData;
