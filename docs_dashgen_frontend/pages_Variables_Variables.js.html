import React, { useEffect } from 'react';
import { Link } from 'react-router-dom';
import Grid from '@material-ui/core/Grid';
import Button from '../../components/Button/Button';
import Navbar from '../../components/Navbar/Navbar';
import CheckboxList from '../../components/CheckboxList/CheckboxList';
import './Variables.css';

/**
 * Na Página de escolha de variáveis é apresentado a lista de variáveis
 * relativa aos painéis selecionados anteriormente pelo o usuário.
 *
 * @interface
 *
 */

function Variables({ selectedPanel, selectedVariable, setSelectedVariable }) {
  useEffect(() => {
    const newState = {};
    Object.keys(selectedVariable).map((item) => {
      if (selectedPanel.indexOf(item) !== -1) { newState[item] = selectedVariable[item]; }
    });
    setSelectedVariable(newState);
  }, []);

  return (
    <>
      {/* Navbar div */}
      <Grid item xs={12} sm={12} md={12} lg={12}>
        <Navbar />
      </Grid>

      <Grid container className="contentBox">
        <Grid item xs={12} sm={12} md={12} lg={12}>
          <h2 className="page-title">Escolha as variáveis dos painéis selecionados</h2>
        </Grid>

        {
          selectedPanel.map((item, index) => (
            <Grid item key={index} className="checkbox-list-container" xs={12} sm={12} md={6} lg={6}>
              <CheckboxList titlePanel={item} selectedVariable={selectedVariable} setSelectedVariable={setSelectedVariable} variablesList={['v1', 'v2', 'v3', 'v4']} />
            </Grid>
          ))
        }

        <Grid className="page-footer" item xs={12} sm={12} md={12} lg={12}>
          <Link to="selecao-paineis"><Button>ETAPA ANTERIOR</Button></Link>
          <Link to="resumo">
            <Button
              disabled={
                (Object.keys(selectedVariable)
                  .filter((item) => selectedVariable[item].length > 0)).length === 0
              }
            >
              PRÓXIMA ETAPA
            </Button>
          </Link>
        </Grid>
      </Grid>
    </>
  );
}

export default Variables;
