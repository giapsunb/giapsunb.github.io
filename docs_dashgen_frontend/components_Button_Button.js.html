import React from 'react';
import './Button.css';

/**
 * Retorna um botão com o estilo padrão da UI
 *
 * @component
 * @example
 * const children = 'ETAPA ANTERIOR'
 * const disable = false
 * return (
 *   <Button>ETAPA ANTERIOR</Button>
 * )
 */

function Button({ children, disabled }) {
  return (
    <button type="button" className="btn-primary" disabled={disabled}>{children}</button>
  );
}

export default Button;
